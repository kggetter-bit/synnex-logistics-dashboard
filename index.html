function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const d = JSON.parse(e.postData.contents); const shName = d.type === 'capacity' ? 'CapacityData' : 'LogisticsData'; let sh = ss.getSheetByName(shName) || ss.insertSheet(shName); if (sh.getLastRow() === 0) { if (d.type === 'capacity') { sh.appendRow(['Timestamp','Year','Month', 'HQ_PalletUse','HQ_PalletCap', 'HQ_ToteUse','HQ_ToteCap', 'HQ_F1','HQ_F2Robot','HQ_F2ASRS', 'HQ_F3','HQ_F4', 'MDC_PalletUse','MDC_PalletCap','MDC_SKU', 'TKS_PalletUse','TKS_PalletCap','TKS_SKU', 'BP_PalletUse','BP_PalletCap','BP_SKU']); } else { sh.appendRow(['Timestamp','Year','Month', 'Warehouse','Revenue','Inventory', 'WorkingDays','COA','Orders','Items', 'CostBKK','CostUPC','DropBKK','DropUPC', 'Storage','PickPack']); } } if (d.type === 'capacity') { sh.appendRow([new Date(),d.yr,d.mo, d.hqPU,d.hqPC,d.hqTU,d.hqTC, d.hqF1,d.hqF2r,d.hqF2a,d.hqF3,d.hqF4, d.mdcPU,d.mdcPC,d.mdcSKU, d.tksPU,d.tksPC,d.tksSKU, d.bpPU,d.bpPC,d.bpSKU]); } else { sh.appendRow([new Date(),d.yr,d.mo,d.wh, d.rev,d.inv,d.wd,d.coa,d.ord,d.itm, d.cbkk,d.cupc,d.dbkk,d.dupc,d.stor,d.pp]); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function doGet() { return ContentService .createTextOutput(JSON.stringify({status:'connected'})) .setMimeType(ContentService.MimeType.JSON); }
